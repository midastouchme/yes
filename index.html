const express = require('express'); const path = require('path'); // To handle file paths const app = express(); // Serve static files from the 'public' folder app.use(express.static(path.join(dirname, '../public'))); // Define a route to send the HTML file app.get('/', (req, res) => { res.sendFile(path.join(dirname, '../public/index.html')); }); module.exports = app;
